    <script>
        const endDate = new Date('June 30, 2027 23:59:59').getTime();
        const startDate = new Date('January 1, 2022 00:00:00').getTime();
        const contractTotalMs = endDate - startDate;
        
        // Mức lương theo năm (Euro)
        const grossSalary = 10000000; // 10 củ Barca trả
        const netSalary = 4800000;    // 4.8 củ Ferran đút túi
        let currentAnnualSalary = grossSalary;
        
        // Trung bình năm có 365.25 ngày (tính cả năm nhuận)
        const daysInYear = 365.25;

        // Xử lý sự kiện đổi loại lương
        const btnGross = document.getElementById('btn-gross');
        const btnNet = document.getElementById('btn-net');
        const salaryTitle = document.getElementById('salary-title');

        btnGross.addEventListener('click', () => {
            currentAnnualSalary = grossSalary;
            btnGross.classList.add('active');
            btnNet.classList.remove('active');
            salaryTitle.textContent = "Tổng tiền Barca đã mất";
        });

        btnNet.addEventListener('click', () => {
            currentAnnualSalary = netSalary;
            btnNet.classList.add('active');
            btnGross.classList.remove('active');
            salaryTitle.textContent = "Tổng tiền Ferran đút túi";
        });

        function update() {
            const now = new Date().getTime();
            const distance = endDate - now;
            const timeElapsedMs = now - startDate;

            // 1. Cập nhật đếm ngược thời gian (dùng textContent tối ưu hơn)
            if (distance > 0) {
                document.getElementById('days').textContent = String(Math.floor(distance / (1000 * 60 * 60 * 24))).padStart(2, '0');
                document.getElementById('hours').textContent = String(Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60))).padStart(2, '0');
                document.getElementById('minutes').textContent = String(Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60))).padStart(2, '0');
                document.getElementById('seconds').textContent = String(Math.floor((distance % (1000 * 60)) / 1000)).padStart(2, '0');
            } else {
                document.getElementById('days').textContent = "00";
                document.getElementById('hours').textContent = "00";
                document.getElementById('minutes').textContent = "00";
                document.getElementById('seconds').textContent = "00";
            }

            // 2. Tính toán lương nhảy liên tục
            if (timeElapsedMs > 0) {
                const salaryPerMillisecond = currentAnnualSalary / (daysInYear * 24 * 60 * 60 * 1000);
                const totalEarned = timeElapsedMs * salaryPerMillisecond;
                document.getElementById('salary').textContent = 
                    '€' + totalEarned.toLocaleString('en-US', { minimumFractionDigits: 2, maximumFractionDigits: 2 });
            }

            // 3. Xử lý thanh tiến trình (Progress Bar)
            if (timeElapsedMs > 0) {
                let percent = (timeElapsedMs / contractTotalMs) * 100;
                if (percent > 100) percent = 100; // Không vượt quá 100% khi hết hạn
                
                document.getElementById('progress-fill').style.width = percent + '%';
                document.getElementById('progress-text').textContent = `Tiến độ hoàn thành hợp đồng: ${percent.toFixed(5)}%`;
            }

            requestAnimationFrame(update);
        }

        requestAnimationFrame(update);
    </script>
